import {
  Component,
  OnDestroy,
  OnInit,
  ViewChild,
  AfterViewInit,
} from '@angular/core';
import { OverviewService } from '../../../services/ticket-overview/overview.service';
import { AppRoute } from '../../../app.routes';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort, Sort, SortDirection } from '@angular/material/sort';
import { ITableParams } from '../../../core/models/base/paged-result.model';
import { TicketOverview } from '../../../core/models/ticket-overview/ticket-overview.model';
import { MatTableDataSource } from '@angular/material/table';
import { BehaviorSubject, merge, Subject, switchMap, tap, timeout } from 'rxjs';
import { GenericFilter } from '../../../core/models/base/filter/generic-filter';
import { FormBuilder, FormGroup } from '@angular/forms';
import { StorageService } from '../../../services/system/storage.service';
import { ChangeRequestorDialogComponent } from '../../dialog/change-requestor-dialog/change-requestor-dialog.component';
import { MatDialog } from '@angular/material/dialog';
import { Item } from '../../../core/models/base/item.interface';
import { AssigmentRequest } from '../../../core/models/userAssignment/assigmentRequest.model';
import { AuthService } from '../../../services/auth.service';
import { LoadingService } from 'src/app/services/system/loading.service';
import { ActivatedRoute, Router } from '@angular/router';
import { AssignUserDialogComponent } from '../../dialog/assign-user-dialog/assign-user-dialog.component';
import { DatePipe } from '@angular/common';
import { RequestorSuggestService } from 'src/app/services/suggest/requestor.suggest.service';
import { EmsUserSuggestService } from 'src/app/services/suggest/ems-user.suggest.service';
import { Permission } from 'src/app/core/models/security/permission.enum';

@Component({
    selector: 'app-overview',
    templateUrl: './overview.component.html',
    styleUrl: './overview.component.scss',
    standalone: false
})
export class OverviewComponent implements OnInit, OnDestroy {
  routes = AppRoute;
  isFilterOpen = false;
  isButtonDisabled = true;

  @ViewChild(MatPaginator) paginator: MatPaginator;
  @ViewChild(MatSort) sort: MatSort;
  tableParams: ITableParams;

  entityVisibleProperties = Object.getOwnPropertyNames(new TicketOverview())
    .filter(
      (x) =>
        x != 'id' &&
        x != 'tenantId' &&
        x != 'ticketStatus' &&
        x != 'ticketCode' &&
        x != 'requesterUserId' &&
        x != 'assignedUserId' 
        //x != 'requesterLoS'
    )
    .map((x) => x.toUpperCase());

  readonly entityFields = [...this.entityVisibleProperties];

  readonly sortings = [...this.entityVisibleProperties];
  searchFormBackUp: any;

  getDisplayedColumns(): string[] {
    return [
      'CHECK',
      'TICKETCODE',
      'TICKETSTATUS',
      ...this.entityFields,
      //'ACTIONS',
    ];
  }

  ticketsId: number[] = [];
  assignedUsersId: number[] = [];
  userAssgmentRequest: AssigmentRequest[] = [];
  requesterUsersId: number[] = [];
  changeRequestorRequest: AssigmentRequest[] = [];

  readonly TABLE_PARAMS_TOKEN = 'overview-tableParams';
  dataSource = new MatTableDataSource<TicketOverview>([]);
  dataTable$: Subject<TicketOverview[]> = new Subject();

  formBuilder: FormBuilder = new FormBuilder();
  searchForm: FormGroup;
  tableCountSubject = new BehaviorSubject<number>(0);

  unsubscribeAll: Subject<void>;

  tenantsItem: Item[] = [];
  statusItem: Item[] = [];
  stageItem: Item[] = [];
  ticketTypesItem: Item[] = [];

  userTerritory: string;

  constructor(
    private dialog: MatDialog,
    private authService: AuthService,
    private storageService: StorageService,
    private loadingService: LoadingService,
    private activatedRoute: ActivatedRoute,
    private overviewService: OverviewService,
    private datePipe: DatePipe,
    private router: Router,
    public requestorSuggestService: RequestorSuggestService,
    public emsUserSuggestService: EmsUserSuggestService,
  ) {
    this.unsubscribeAll = new Subject();

    this.tableParams = {
      pageSize: 0,
      sortDirection: 'asc',
      sortBy: 'ID',
      filterForm: {},
    };
  }

  ngOnInit(): void {    
    this.userTerritory = this.authService.currentUser.tenants[0]?.description;

    this.statusItem = this.activatedRoute.snapshot.data['status'] as Item[];
    this.stageItem = this.activatedRoute.snapshot.data['stages'] as Item[];
    this.ticketTypesItem = this.activatedRoute.snapshot.data['types'] as Item[];
    this.tenantsItem = this.activatedRoute.snapshot.data['tenant'] as Item[];

    const filterForm = JSON.parse(
      this.storageService.getItem(this.TABLE_PARAMS_TOKEN)!
    )?.filterForm;

    this.searchForm = this.formBuilder.group({
      tenantId: [filterForm?.tenantId || null],
      prid: [filterForm?.prid || null],
      partyName: [filterForm?.partyName || null],
      ticketTypeIds: [filterForm?.ticketTypeIds || null],
      stageIds: [filterForm?.stageIds || null],
      statusIds: [filterForm?.statusIds || null],
      requesterId: [filterForm?.requesterId || null],
      assingmentUserId: [filterForm?.assingmentUserId || null],
      ticketCode: [filterForm?.ticketCode || null],
      salesforceAccountId: [filterForm?.salesforceAccountId || null]
    });

    this.requestorSuggestService.emitUpdateSignal();
    this.emsUserSuggestService.emitUpdateSignal();

    this.getTicketsOverview();
  }

  ngOnDestroy() {
    this.unsubscribeAll.next();
    this.unsubscribeAll.complete();
  }

  ngAfterViewInit(): void {
    this.loadData();

    this.sort.sortChange.subscribe(() => (this.paginator.pageIndex = 0));
    this.searchForm.valueChanges.subscribe({
      next: () => {
        this.paginator.pageIndex = 0
        this.saveTableParams();
      }
    });
    
    merge(this.sort.sortChange, this.paginator.page)
      .pipe(
        tap((_) => {
          this.loadData();
        })
      )
      .subscribe();

    this.enableFormFilters();
  }

  getTicketsOverview() {   
    this.dataTable$
      .pipe(
        switchMap(() => {          
          setTimeout(() => {
            this.loadingService.setSpinner(true);
          }, 10);

          this.searchFormBackUp = this.searchForm.value;
          const filter = new GenericFilter().generate(
            `${this.tableParams.sortBy} ${this.tableParams.sortDirection}`,
            this.paginator,
            this.searchFormBackUp
          );

          return this.overviewService.getTicketsOverview(filter);
        })
      )
      .subscribe({
        next: (response) => {
          this.dataSource.data = response.results;
          this.tableCountSubject.next(response.rowCount);
          
          this.searchForm.patchValue(this.searchFormBackUp, { emitEvent: false });
          this.requestorSuggestService.emitUpdateSignal();
          this.emsUserSuggestService.emitUpdateSignal();
          
          this.saveTableParams();
        },
      });
  }

  saveTableParams() {
    // Save filters in Local Storage
    this.tableParams.pageSize = this.paginator.pageSize;
    this.tableParams.filterForm = this.searchForm.getRawValue();
    this.storageService.setItem(
      this.TABLE_PARAMS_TOKEN,
      JSON.stringify(this.tableParams)
    );
  }

  loadData() {
    this.dataTable$.next([]);
  }

  clearFilters() {
    this.searchForm.reset();
    this.loadData();
  }

  applyFilters() {
    this.loadData();
  }

  enableFormFilters() {
    this.searchForm.valueChanges.subscribe({
      next: (x) => {
        this.tableParams.filterForm.requesterId = this.searchForm.get('requesterId').value;
        this.tableParams.filterForm.assignedUsersId = this.searchForm.get('assignedUsersId').value;
        this.storageService.setItem(
          this.TABLE_PARAMS_TOKEN,
          JSON.stringify(this.tableParams)
        );
        this.loadData();
      },
    });
  }

  setSortingOrder(sortDirection: SortDirection) {
    this.tableParams.sortDirection = sortDirection;
    this.loadData();
  }

  setSortingBy(sortBy: string) {
    this.tableParams.sortBy = sortBy;
    this.loadData();
  }

  setMatSorting(sort: Sort) {
    
    if (this.tableParams.sortBy !== sort.active) {
      this.setSortingBy(sort.active);
    }
    
    // if (this.tableParams.sortDirection !== sort.direction) {
      console.log(sort.direction);
      sort.direction = sort.direction || this.tableParams.sortDirection === 'asc'
          ? 'desc'
          : 'asc';
      this.setSortingOrder(sort.direction);
    // }
  }

  toggleFilter() {
    this.isFilterOpen = !this.isFilterOpen;
  }

  openChnageRequestorDialog(item: TicketOverview | null): void {
    if (item != null) {
      this.ticketsId = [];
      this.assignedUsersId = [];
      this.ticketsId.push(item.id!);
      this.assignedUsersId.push(item.assignedUserId);
      this.changeRequestorRequest.push({
        userId: item.assignedUserId,
        ticketId: item.id!,
        tenantId: item.tenantId,
        userName: item.requesterUserEmail
      });
    }

    const dialogRef = this.dialog.open(ChangeRequestorDialogComponent, {
      width: '600px',
      position: { top: '10%' },
      data: { userRequestor: this.changeRequestorRequest },
    });

    dialogRef.afterClosed().subscribe((result) => {
      this.ticketsId = [];
      this.assignedUsersId = [];
      this.requesterUsersId = [];
      this.changeRequestorRequest = [];
      this.userAssgmentRequest = [];
      this.loadData();
    });
  }

  openAssignmentDialog(item: TicketOverview | null): void {
    if (item != null) {
      this.ticketsId = [];
      this.assignedUsersId = [];
      this.ticketsId.push(item.id!);
      this.assignedUsersId.push(item.assignedUserId);
      this.userAssgmentRequest.push({
        userId: item.assignedUserId,
        ticketId: item.id!,
        tenantId: item.tenantId,
        userName: item.assignedUserEmail
      });
    }

    const dialogRef = this.dialog.open(AssignUserDialogComponent, {
      width: '600px',
      position: { top: '10%' },
      data: { userAssing: this.userAssgmentRequest },
    });

    dialogRef.afterClosed().subscribe((result) => {
      this.ticketsId = [];
      this.assignedUsersId = [];
      this.requesterUsersId = [];
      this.changeRequestorRequest = [];
      this.userAssgmentRequest = [];
      this.loadData();
    });
  }

  itemSelectionChanged(selected: boolean, item: TicketOverview) {
    var assignedUserId = item.assignedUserId;
    var requesterUserId = item.requesterUserId;
    
    if (
      selected &&
      //!this.assignedUsersId.includes(assignedUserId) &&
      //!this.requesterUsersId.includes(requesterUserId) &&
      !this.ticketsId.includes(item.id)
    ) {
      //this.assignedUsersId.push(assignedUserId);
      //this.requesterUsersId.push(requesterUserId);
      this.ticketsId?.push(item.id!);
      this.userAssgmentRequest.push({
        userId: assignedUserId,
        ticketId: item.id!,
        tenantId: item.tenantId,
        userName: item.assignedUserEmail
      });
      this.changeRequestorRequest.push({
        userId: assignedUserId,
        ticketId: item.id!,
        tenantId: item.tenantId,
        userName: item.requesterUserEmail
      });
    }
    if (
      !selected &&
      //this.assignedUsersId.includes(assignedUserId) &&
      //this.requesterUsersId.includes(requesterUserId) &&
      this.ticketsId.includes(item.id!)
    ) {
      // this.assignedUsersId = this.assignedUsersId.filter(
      //   (x) => x != assignedUserId
      // );
      // this.requesterUsersId = this.requesterUsersId.filter(
      //   (x) => x != requesterUserId
      // );
      this.ticketsId = this.ticketsId?.filter((x) => x != item.id);
      this.userAssgmentRequest = this.userAssgmentRequest.filter(
        (x) => x.userId != assignedUserId && x.ticketId != item.id
      );
      this.changeRequestorRequest = this.changeRequestorRequest.filter(
        (x) => x.userId != requesterUserId && x.ticketId != item.id
      );
    }
  }

  isItemSelected(item: TicketOverview): boolean {
    return (
      //this.assignedUsersId.some((x) => x == item.assignedUserId) &&
      //this.requesterUsersId.some((x) => x == item.requesterUserId)
      this.ticketsId.some((x) => x == item.id)
    );
  }

  setAllItemsChecked(checked: boolean) {
    this.assignedUsersId = [];
    this.requesterUsersId = [];
    this.ticketsId = [];
    this.changeRequestorRequest = [];
    this.userAssgmentRequest = [];

    if (checked) {
      this.assignedUsersId = this.dataSource.data.map((x) => x.assignedUserId);
      this.requesterUsersId = this.dataSource.data.map(
        (x) => x.requesterUserId
      );
      this.ticketsId = this.dataSource.data.map((x) => x.id!);
      this.changeRequestorRequest = this.dataSource.data.map((x) => ({
        userId: x.assignedUserId,
        ticketId: x.id!,
        tenantId: x.tenantId,
        userName: ''
      }));
      this.userAssgmentRequest = this.dataSource.data.map((x) => ({
        userId: x.requesterUserId,
        ticketId: x.id!,
        tenantId: x.tenantId,
        userName: ''
      }));
    } else {
      this.assignedUsersId = [];
      this.requesterUsersId = [];
      this.ticketsId = [];
      this.changeRequestorRequest = [];
      this.userAssgmentRequest = [];
    }
  }

  allItemsChecked(): boolean {
    const dataSourceIds = this.dataSource.data.map((x) => x.assignedUserId);
    const commonIds = this.assignedUsersId.filter((x) =>
      dataSourceIds.includes(x)
    );
    // console.log(commonIds.length >= this.dataSource.data.length &&
    //   this.dataSource.data.length > 0)
    // Check length of the intersection
    return (
      commonIds.length >= this.dataSource.data.length &&
      this.dataSource.data.length > 0
    );
  }

  someItemsChecked(): boolean {
    const dataSourceIds = this.dataSource.data.map((x) => x.assignedUserId);
    const commonIds = this.assignedUsersId.filter((x) =>
      dataSourceIds.includes(x)
    );
    return (
      0 < commonIds.length && commonIds.length < this.dataSource.data.length
    );
  }

  enableUserAssignmentButtos(): boolean {
    if (this.ticketsId != undefined && this.ticketsId.length > 0) return false;
    else return true;
  }

  enableTableButtons(): boolean {
    if (this.ticketsId.length <= 0) return false;
    else return true;
  }

  getItemValue(item: TicketOverview, key: string): string {
    const formatter: any = {
      CREATIONDATE: (d: Date) => this.datePipe.transform(d,'d-MMM-y', null, 'es-AR'),
      MODIFICATIONDATE: (d: Date) => this.datePipe.transform(d,'d-MMM-y', null, 'es-AR'),
    
      //formateo de reason
      REASON: (value: string) => {
        if (!value) return '';
        const prefix = value.split(' ')[0];
        return `Ticket ${prefix}`
      }
    };


    const searchedKey = Object.keys(item).find(
      (x) => x.toLowerCase() === key.toLowerCase()
    );
    const searchedValue = (item as any)[searchedKey];

    return key in formatter ? formatter[key](searchedValue) : searchedValue
  }

  getFormByStaus(item: TicketOverview) {
    if (item.ticketStage == 'Entity Creation') {
      if (item.ticketStatus == 'Open') {
        this.router.navigate([`ems/entity/create/approve`, item.id]);
      } else {
        this.router.navigate([`ems/entity/create`, item.id]);
      }
    }
    if (item.ticketStage == 'Entity Update') {
      if (item.ticketStatus == 'Open') {
        this.router.navigate([`ems/entity/update/approve`, item.id]);
      } else {
        this.router.navigate([`ems/entity/update`, item.id]);
      }
    }
    if (item.ticketStage == 'Hierarchy Creation/Update') {
      if (item.ticketStatus == 'Open') {
        this.router.navigate([`ems/hierarchy/approve`, item.id]);
      } else {
        this.router.navigate([`ems/hierarchy/`, item.id]);
      }
    }
    if (item.ticketStage == 'Entity Verification') {
        this.router.navigate([`ems/verification/`, item.id]);
    }
  }

  canAssign(){
    const permissions = [Permission.HierarchyApproval,Permission.CreateEntityApproval,Permission.UpdateEntityApproval,Permission.VerificationApproval];
    return this.authService.isAuthorized(permissions);
  }
}
