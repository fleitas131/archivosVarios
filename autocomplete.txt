ALTER VIEW [EMS].[vTicketInfo] AS
SELECT
    t.CreationDate,
    t.ModificationDate,
    t.TicketCode,
    ent.Prid,
    sa.SalesforceAccountId,
    sa.PartyId,
    ent.PartyName,
    tt.Description AS TicketType,
    req.Email AS RequestedBy,
    req.LoS AS LOS,
    asg.Email AS AssignedTo,
    stg.Description AS TicketStage,
    s.Description AS TicketStatus,
    tenant.CountryISOName AS Tenant
FROM ems.Ticket t
JOIN App.Tenant tenant ON t.TenantId = tenant.Id
JOIN ems.tickettype tt ON t.tickettypeid = tt.id
JOIN ems.Status s ON t.StatusId = s.Id
JOIN ems.Stage stg ON t.StageId = stg.Id
JOIN ems.Entity ent ON t.EntityId = ent.Id
JOIN ems.[user] req ON req.Id = t.RequesterUserId
LEFT JOIN ems.SalesforceAccount sa ON t.SalesforceAccountId = sa.Id
LEFT JOIN ems.[user] asg ON asg.Id = t.AssignedUserId;
GO
