  activateFragmentNavigation() {
    if (!this.historicalDataTabGroup) return;
    this.activatedRoute.fragment.subscribe(fragment => {
      fragment ??= Stage[this.createMode ? Stage.EntityCreation : Stage.EntityUpdate];
      
      const tabs = this.historicalDataTabGroup._tabs.toArray();
      const fragmentTab = tabs.find(x => x.textLabel.replace(/ /gm, '') == fragment);
      setTimeout(() => {
      this.selectedTabIndex = tabs.indexOf(fragmentTab);
      this.cdr.detectChanges(); // <-- fuerza la actualización
       }, 500);
    });
  }


  ngOnInit(): void {
    this.loadingService.setSpinner(false, '', '');
    
    this.ticketForm = this.generateTicketForm();

    this.loadSnapshotData();

    this.linkSuggestServices();
    this.setMode();

    //revisamos si tiene EMS User asignado
    this.hasEMSUser();

    //revisamos si user no es requester
    this.isNotRequester();

    // this.ticketForm.valueChanges.subscribe(v => {
    //   console.log('FULL FORM VALUE', v)
    // })

    this.buildHistoricalDataForm();    

    this.applyContextValidators();

    if ((!this.updateMode  || (this.createMode && !this.approveMode)) && !this.isVerificationTicket()) {
      this.ticketForm.get(['generalInformation', 'prid']).disable();
    }

    if ((this.status == Status.CANCELED || this.status == Status.CLOSED) || 
        (this.canSubmit() && this.status == Status.OPEN ) ||
        (this.canApprove() && this.status == Status.PENDING)) {
      this.ticketForm.disable({ emitEvent: false });
      this.copyAddressEnabled = false;
    }

    if(this.validatePrid()){
      this.ticketForm.get(['generalInformation', 'prid']).enable();
      this.ticketForm.get(['generalInformation', 'prid']).markAsTouched();
      this.ticketForm.get(['generalInformation', 'prid']).addValidators(Validators.required);
    }

    if (this.status == Status.CANCELED || this.status == Status.CLOSED) {
      this.showCommentsAndAtth = false;
    }

    // ----------------------------
    // Forms Events
    // ----------------------------

    // On clientType change
    this.ticketForm
      .get(['header', 'clientType'])
      .valueChanges
      .subscribe(() => this.applyContextValidators());
    
    // On registered address country change
    this.ticketForm
      .get(['registeredAddress', 'countryId'])
      .valueChanges
      .subscribe((countryId) => {
        this.registeredStateSuggestService.countryId = countryId;
        this.ticketForm.get(['registeredAddress', 'stateId']).reset();
        setTimeout(() => {
          this.ticketForm.get(['registeredAddress', 'stateId']).markAsTouched();  
        }, 500);
      });
    
    // On operating address country change
    this.ticketForm
      .get(['operatingAddress', 'countryId'])
      .valueChanges
      .subscribe((countryId) => {
        this.operatingStateSuggestService.countryId = countryId;
        this.ticketForm.get(['operatingAddress', 'stateId']).reset();
        setTimeout(() => {
          this.ticketForm.get(['operatingAddress', 'stateId']).markAsTouched();
        }, 500);
      });
    
    //On PRID change
    const currentUrl = this.router.url;

    if (currentUrl.includes('/ems/entity/update/verification'))
    {
        this.ticketForm.get(['generalInformation', 'prid']).valueChanges.pipe(
        filter(() => !this.createMode && (this.ticketId <= 0)),
        startWith(''),
        distinctUntilChanged(),
        // delay emits
        debounceTime(1000),
        switchMap(prid => {
          if (prid !== '' && typeof prid == 'string') {
            this.loadingService.setSpinner(true);
            return this.entityService.getByPrid(prid, this.ticketTypeId);
          }
          else return of(null);
        }),
        filter(entity => entity != null)
      ).subscribe({
        next: entity => {
          if(entity?.generalInformation?.prid != '' && entity?.generalInformation?.prid != null){
            this.patchEntityByPrid(entity);
          }
        }
      });
    }
    

    // On Ticket status change
    this.status$.pipe(
      filter(s => s == Status.OPEN && (this.createMode || this.updateMode))
    ).subscribe(() => {
      this.copyAddressEnabled = false;
      this.ticketForm.disable({ emitEvent: false });
    })

    // No validation is performed in approve mode
    this.status$.pipe(
      filter(s => s == Status.OPEN && this.approveMode)
    ).subscribe(() => {
      this.clearEntityFormValidators()
    })

    // On any Registered form values changes
    // Enabled address copy if there's a value
    this.ticketForm.get('registeredAddress')
      .valueChanges
      .subscribe(() => {
        this.checkCopyAddressEnable();
      })

    //console.log('create mode', this.createMode);
    //console.log('update mode', this.updateMode);

    // On Party Id change
    this.ticketForm.get(['generalInformation', 'partyTypeId'])
      .valueChanges
      .subscribe(async (partyTypeId:number) => {
        const emailControl = this.ticketForm.get(['generalInformation', 'email']);
        const EXTERNAL_PERSON = 'External Person';

        const item = this.selectItem
          .generalInformation
          .partyTypeId
          .find(x => x.id == partyTypeId)

        if (this.createMode && item.description == EXTERNAL_PERSON) {
          emailControl.addValidators(Validators.required);
        } else if (this.updateMode) {
          emailControl.removeValidators(Validators.required);
        } else {
          emailControl.removeValidators(Validators.required);
        }
        
        emailControl.updateValueAndValidity();

        // Updates service partyType for filtering organization types
        this.ticketForm.get(['generalInformation', 'organisationTypeId']).reset();
        this.ticketForm.get(['generalInformation', 'organisationTypeId']).markAsTouched();
        this.organizationTypeSuggestService.partyType = item.description;
        this.selectItem.generalInformation.organisationTypeId = await lastValueFrom(this.organizationTypeSuggestService.getAllItems());
      })


    if ((this.updateMode && (this.status == Status.PENDING) && this.ticketForm.get(['header','ticketId']).value > 0)) {
      this.ticketForm.get(['generalInformation', 'prid']).disable();
    }

    setTimeout(() => {
      this.markFormAsTouched();
    }, 1000);

    //console.log(this.ticketForm);
    //console.log('ticket status',this.status);
  }

  numbersOnly(event: Event): void {
    const input = event.target as HTMLInputElement;
    let cleaned = input.value.replace(/\D+/g, '');
    cleaned = cleaned.slice(0,9);
    input.value = cleaned;
    this.ticketForm.get(['generalInformation', 'prid']).setValue(cleaned, { emitEvent: false });
  }

  hasEMSUser(): boolean{
    let check: boolean = false;
    
    const assigneeEmail = this.ticketForm.get(['header', 'assignee']).value;
    const emsUserEmail = this.authService.currentUser.email;

    if (assigneeEmail != null && assigneeEmail == emsUserEmail && this.canApprove() && this.status != Status.CANCELED && this.status != Status. CLOSED) {
      check = true;
    }

    return check;
  }

  //si es ems user puede cancelar el ticket
  isNotRequester(): boolean{
    let check: boolean = false;
    
    const isNotReq = this.authService.currentUser.roles[0].id;

    if (isNotReq != 1 && this.hasEMSUser())
    {
      this.approveMode = true;
      check = true;
    }

    return check;
  }

  buildHistoricalDataForm() {
    const ticketHistory = this.activatedRoute.snapshot.data['ticketHistory'] as Ticket[];

    //console.log('ticket history', ticketHistory);

    if (!Array.isArray(ticketHistory) || ticketHistory.length === 0) {
      return;
    }

    for (const item of ticketHistory) {
      const historicalTicketForm = this.generateTicketForm(true);

      historicalTicketForm.get('header').patchValue(item.header, { emitEvent: false }); 
      historicalTicketForm.get('generalInformation').patchValue(item.generalInformation, { emitEvent: false }); 
      historicalTicketForm.get('operatingAddress').patchValue(item.operatingAddress, { emitEvent: false }); 
      historicalTicketForm.get('registeredAddress').patchValue(item.registeredAddress, { emitEvent: false }); 

      historicalTicketForm.disable({ emitEvent: false });

      this.ticketFormHistory.push(historicalTicketForm);
    }
  }

    this.ticketForm.get('registeredAddress').patchValue(entity.registeredAddress, { emitEvent: false }); 

    this.restoreAutocompleteFields(entity);

    this.updateSuggestControls();
    //this.markFormAsTouched();

    // Added some delay for 
    // avoiding race condition
    // setTimeout(() => {
    //   this.loadingService.setSpinner(true);
    // }, 100);
  }

  navigateToHistoricalData(event: MatTabChangeEvent) {
    const stage = Stage[
      event.tab.textLabel
        .replace(/ /gm, '')
        .replace('/', '') as keyof typeof Stage
    ];
    
    let linkObj = null;
    
    switch (stage) {
      case Stage.HierarchyCreationUpdate: linkObj = [AppRoute.HIERARCHY, this.ticketId]; break;          
      case Stage.EntityCreation: 
        // if (this.status == Status.OPEN) {
        //   linkObj = [AppRoute.ENTITY, AppRoute.CREATE, AppRoute.APPROVE, this.ticketId];
        // } else {
          linkObj = [AppRoute.ENTITY, AppRoute.CREATE, this.ticketId]; 
        //}
        break;
      case Stage.EntityVerification: linkObj = [AppRoute.VERIFICATION, this.ticketId]; break;
      case Stage.EntityUpdate: 
        if (this.status == Status.OPEN) {
          linkObj = [AppRoute.ENTITY, AppRoute.UPDATE, AppRoute.APPROVE, this.ticketId];
        } else {
          linkObj = [AppRoute.ENTITY, AppRoute.UPDATE, this.ticketId]; 
        }
        break;
        
      default: linkObj = null; break;
    }
    
    linkObj.unshift(AppRoute.EMS);
    linkObj.unshift(AppRoute.ROOT);
    
    this.router.navigate(linkObj, {
      fragment: Stage[stage]
    });
  }

  clearEntityFormValidators(form: FormGroup = this.ticketForm) {
    (<any>Object).values(form.controls).forEach((control: any) => {
      if (control instanceof FormGroup) {
        this.clearEntityFormValidators(control);
      }
      else if (control !== this.ticketForm.get(['generalInformation', 'prid'])) {
        (<FormControl>control).clearValidators();
      }
    });
  }

  applyContextValidators() {
    if (this.status == Status.OPEN) {
      this.clearEntityFormValidators();
      return;      
    }

    for (const subform in this.ticketForm.controls) {
      for (const controlKey in this.ticketForm.get(subform).value) {
        const validators = this.getContextValidators(subform, controlKey);

        if (validators != null) {
          const control = this.ticketForm.get([`${subform}`, `${controlKey}`]);
          control.clearValidators();          
          control.addValidators(validators);
          control.updateValueAndValidity({ emitEvent: false });
        }
      }
    }
  }

  getContextValidators(subformKey: string, controlKey: string): ValidatorFn[] { 
    const clientType = this.ticketForm.get(['header', 'clientType']).value as boolean;
    const ticketType = <TicketType>this.ticketTypeId;

    const validatorSwitch = (ignoreValue: boolean = false, requiredValue: boolean = true) => {
      return !ignoreValue && clientType == requiredValue ?
        [Validators.required] :
        [];
    };
    
    const requireValue = () => {
      return (ticketType == TicketType.CreateEntity) || (ticketType == TicketType.EntityVerification) || (ticketType == TicketType.SalesforceOpportunity) || (ticketType == TicketType.EntityCreationVerifcation) ?
        [Validators.required] :
        [];
    };

    const valueIgnored = [TicketType.UpdateEntity].includes(ticketType);

    const validators: any = {
      generalInformation: {
        prid: ((ticketType == TicketType.EntityVerification) || (ticketType == TicketType.UpdateEntity)) ? [Validators.required] : [],
        organisationTypeId: requireValue(),
        identifierTypeId: requireValue(),
        identifierValue: requireValue(),
        pwcSubindustryId: validatorSwitch(valueIgnored),
      },
      registeredAddress: {
        countryId: requireValue(),
        stateId: requireValue(),
        city: requireValue(),
        addressLine1: validatorSwitch(valueIgnored),
      },
      operatingAddress: {
        countryId: validatorSwitch(valueIgnored),
        stateId: validatorSwitch(valueIgnored),
        city: validatorSwitch(valueIgnored),
        addressLine1: validatorSwitch(valueIgnored),
      }
    }

    if (!(subformKey in validators)) return null;
    if (!((controlKey in validators[`${subformKey}`]))) return null;

    return validators[`${subformKey}`][`${controlKey}`];
  }

  getTextControlInputType(subformName: string, controlName: string) {
    const defaultType = 'text';
    
    return subformName == 'generalInformation' && controlName == 'email' ?

      defaultType;
  }

  getFormControlLabel(subformName: string, controlName: string): string {
    var subformKey = subformName
      .split(/(?=[A-Z])/)
      .join('-')
      .toUpperCase();
    
    var controlKey = controlName
      .split(/(?=[A-Z])/)
      .join('-')
      .toUpperCase();
    
    if (subformKey.includes('ADDRESS')) subformKey = 'ADDRESS';
    
    return `EMS.ENTITY-EDITOR.${subformKey}.${controlKey}`;
  }

  getSubformTitle(controlName: string): string {
    var key = controlName
      .split(/(?=[A-Z])/)
      .join('-')
      .toUpperCase();
        
    return !key.includes('ADDRESS') ?
      `EMS.ENTITY-EDITOR.${key}.TITLE` :
      `EMS.ENTITY-EDITOR.ADDRESS.TYPE.${key}`;
  }

  getFormControlPlaceholder(subformName: string, controlName: string): string {
    var subformKey = subformName
      .split(/(?=[A-Z])/)
      .join('-')
      .toUpperCase();
    
    var controlKey = controlName
      .split(/(?=[A-Z])/)
      .join('-')
      .toUpperCase();
    
    if (subformKey.includes('ADDRESS')) subformKey = 'ADDRESS';
    
    return `EMS.ENTITY-EDITOR.${subformKey}.PLACEHOLDER.${controlKey}`;
  }

  getSuggestService(subformKey: string, controlKey: string): Observable<ISuggestService> {
    const defaultType: ISuggestService = null;

    const defaultType: Item[] = [{id: 0, description: 'NOT-IMPLEMENTED'}];

    if (!(subformKey in this.selectItem)) return of(defaultType);
    if (!((controlKey in (this.selectItem as any)[`${subformKey}`]))) return of(defaultType);

  getControlType(subformKey: string, controlKey: string): string {
    const defaultType = 'text';

    const controlTypes = [this.suggestItem, this.selectMultipleItem, this.selectItem];

    for (const controlType of controlTypes) {
      if (!(subformKey in controlType)) continue;
      if (!((controlKey in (controlType as any)[`${subformKey}`]))) continue;
      
      return controlType.typeName;
    }

    return defaultType;
  }

  }

  private getTicketFromForm(): Ticket {
    this.updateAutocompleteValues();
    
    let ticket = Object.assign(
      new Ticket(),
      this.ticketForm.getRawValue()
    ) as Ticket;

    ticket.header.sourceId = this.ticketSourceId;
    ticket.header.ticketTypeId = this.ticketTypeId;
    ticket.header.sourceDesc = this.sourceDesc;
    ticket.header.ticketTypeDesc = this.ticketTypeDesc;
    ticket.header.fileIds = this.attachmentFileIds;
    ticket.header.commentIds = this.commentIds;

    return ticket;
  }

  updateAutocompleteValues() {
        const registeredCountry = this.ticketForm.get(['registeredAddress','countryId'])?.value;
    if (registeredCountry != null) {
      const registeredCountryId = registeredCountry.id;
      this.ticketForm.get(['registeredAddress','countryId']).setValue(registeredCountryId, { emitEvent: false });
    }
    
    const registeredState = this.ticketForm.get(['registeredAddress','stateId'])?.value;
    if (registeredState != null) {
      const registeredStateId = registeredState.id;
      this.ticketForm.get(['registeredAddress','stateId']).setValue(registeredStateId, { emitEvent: false });
    }
    
    const operatingCountry = this.ticketForm.get(['operatingAddress','countryId'])?.value;
    if (registeredState != null) {
      const operatingCountryId = operatingCountry.id;
      this.ticketForm.get(['operatingAddress','countryId']).setValue(operatingCountryId, { emitEvent: false });
    }

    const operatingState = this.ticketForm.get(['operatingAddress','stateId'])?.value;
    if (operatingState != null) {
      const operatingStateId = operatingState.id;
      this.ticketForm.get(['operatingAddress','stateId']).setValue(operatingStateId);
    }

    const identifierType = this.ticketForm.get(['generalInformation','identifierTypeId'])?.value;
    if (identifierType != null)
    {
      const identifierTypeId = identifierType.id;
      this.ticketForm.get(['generalInformation','identifierTypeId']).setValue(identifierTypeId, { emitEvent: false });
    }
    
    const stockExchange = this.ticketForm.get(['generalInformation','stockExchangeId'])?.value;
    if (stockExchange != null) {
      const stockExchangeId = stockExchange.id;
      this.ticketForm.get(['generalInformation','stockExchangeId']).setValue(stockExchangeId, { emitEvent: false });
    }
    
    const subIndustry = this.ticketForm.get(['generalInformation','pwcSubindustryId'])?.value;
    if (subIndustry != null) {
      const subIndustryId = subIndustry.id;

    }
  }

  async submit() {
    let ticket = this.getTicketFromForm();

    this.ticketForm.updateValueAndValidity({ emitEvent: false });
    if (this.ticketForm.invalid) {
      this.toastrService.error(this.translate.instant('TOASTR.ENTITY-ERROR'))
      return;
    }
    
    const confirm = await lastValueFrom(
      this.dialogService.openConfirmDialog(this.translate.instant('TOASTR.CONFIRMATION-MESSAGE'), '').afterClosed()
    );
    if (!confirm) return;

    this.updateAutocompleteValues();

    this.loadingService.setSpinner(true);
    const result = await lastValueFrom(
      this.ticketService.submit(ticket)
    );

    this.router.navigate(['/', AppRoute.OVERVIEW]);    
    this.ticketId = result.header.ticketId;
    this.status = result.header.status;

    //this.ticketForm.get('header').patchValue(result.header, { emitEvent: false });
    
  }

  async accept() {
    let ticket = this.getTicketFromForm();

    this.ticketForm.updateValueAndValidity();
    if (this.ticketForm.invalid) {
      this.toastrService.error(this.translate.instant('TOASTR.ENTITY-ERROR'))
      return;
    }

    const confirm = await lastValueFrom(
      this.dialogService.openConfirmDialog(this.translate.instant('TOASTR.APPROVAL-MESSAGE'), '').afterClosed()
    );
    if (!confirm) return;
   
    const sourceDesc = (ticket?.header?.sourceDesc == 'EMS Tool' && (ticket.header.ticketTypeDesc != 'Entity Verification') && (ticket.header.ticketTypeDesc != 'Entity Creation & Verification') )
     ? 'EMS Tool' : 'Automatico';


    this.loadingService.setSpinner(true);
    var result = await lastValueFrom(
      this.ticketService.accept(ticket, sourceDesc)
    );

    if ((result.generalInformation.reason)) {
      this.toastrService.warning(result.generalInformation.reason);
    } else {
      this.router.navigate(['/', AppRoute.OVERVIEW]);
    }

    //this.router.navigate(['/', this.ticketId ? AppRoute.OVERVIEW : AppRoute.DASHBOARD]);
  }

  async reject() {
    let ticket = this.getTicketFromForm();

    this.ticketForm.updateValueAndValidity();
    // if (this.ticketForm.invalid) {
    //   this.toastrService.error(this.translate.instant('TOASTR.ENTITY-ERROR'))
    //   return;
    // }
    
    const dialogData = await lastValueFrom(
      this.dialogService.openRejectDialog(this.translate.instant('TOASTR.REJECT-MESSAGE'), '').afterClosed()
    );
    if (!dialogData.rejected) return;
    
    this.loadingService.setSpinner(true);
    const result = await lastValueFrom(
      this.ticketService.reject(ticket, dialogData.rejectionMessage)
    );

    this.router.navigate(['/', AppRoute.OVERVIEW]);
  }

  canSubmitEMSTeamMember(): boolean {
    const isRequester = this.authService.currentUser.roles.some(
      (role) => role.description == Role.Requestor,
    );
    const isImplementerHierachy = this.authService.currentUser.roles.some(
      (role) => role.description == Role.Implementer_Hierarchy,
    );
    const isIndependenceSpecialist = this.authService.currentUser.roles.some(
      (role) => role.description == Role.Independence_Specialist,
    );

    return (
      isRequester &&
      (isImplementerHierachy || isIndependenceSpecialist) &&
      (this.authService.currentUser.tenantId == 1 ||
        this.authService.currentUser.tenantId == 4)
    ); //Solo aplica para ARG o VEN
  }

  private restoreAutocompleteFields(entiy: IEntity): void {
    //registered country
    //const registeredCountryId = this.ticketForm.get(['registeredAddress', 'countryId']).value;
   if (
        entiy.registeredAddress &&
        Object.values(entiy.registeredAddress).some(v => v !== null && v !== undefined && v !== '')
      ) {
      const registeredCountryId = entiy.registeredAddress.countryId;
      if (registeredCountryId) {
        this.registeredCountrySuggestService.getSuggestPaginated(registeredCountryId.toString(), 0, 1)
        .subscribe(items => {
          const found = items.find(x =>x.id === registeredCountryId);
          if (found) {
            this.ticketForm.get(['registeredAddress', 'countryId']).setValue(found, { emitEvent: false });
          }
        })
      }

      //registered state
      const registeredStateId = this.ticketForm.get(['registeredAddress', 'stateId']).value;
      if (registeredStateId) {
        this.registeredStateSuggestService.getSuggestPaginated(registeredStateId.toString(), 0, 1, registeredCountryId)
        .subscribe(items => {
          const found = items.find(x =>x.id === registeredStateId);
          if (found) {
            this.ticketForm.get(['registeredAddress', 'stateId']).setValue(found, { emitEvent: false });
          }
        })
      }
    }

  if (
        entiy.operatingAddress &&
        Object.values(entiy.operatingAddress).some(v => v !== null && v !== undefined && v !== '')
      ) {
      //operating country
      const operatingCountryId = this.ticketForm.get(['operatingAddress', 'countryId']).value;
      if (operatingCountryId) {
        this.operatingCountrySuggestService.getSuggestPaginated(operatingCountryId.toString(), 0, 1)
        .subscribe(items => {
          const found = items.find(x =>x.id === operatingCountryId);
          if (found) {
            this.ticketForm.get(['operatingAddress', 'countryId']).setValue(found, { emitEvent: false });
          }
        })
      }

      //operating state
      const operatingStateId = this.ticketForm.get(['operatingAddress', 'stateId']).value;
      if (operatingStateId) {
        this.operatingStateSuggestService.getSuggestPaginated(operatingStateId.toString(), 0, 1, operatingCountryId)
        .subscribe(items => {
          const found = items.find(x =>x.id === operatingStateId);
          if (found) {
            this.ticketForm.get(['operatingAddress', 'stateId']).setValue(found, { emitEvent: false });
          }
        })
      }
    }

    //identifier
    const identifierId = this.ticketForm.get(['generalInformation','identifierTypeId']).value;
    if (identifierId) {
      this.identifierTypeSuggestService.getSuggestPaginated(identifierId.toString(), 0, 1)
      .subscribe(items => {
        const found = items.find(x =>x.id === identifierId);
        if (found) {
          this.ticketForm.get(['generalInformation','identifierTypeId']).setValue(found, { emitEvent: false });
        }
      })
    }

    //subindustry
    const subindustryId = this.ticketForm.get(['generalInformation','pwcSubindustryId']).value;
    if (subindustryId) {
      this.subindustrySuggestService.getSuggestPaginated(subindustryId.toString(), 0, 1)
      .subscribe(items => {
        const found = items.find(x =>x.id === subindustryId);
        if (found) {
          this.ticketForm.get(['generalInformation','pwcSubindustryId']).setValue(found, { emitEvent: false });
        }
      })
    }
  }
