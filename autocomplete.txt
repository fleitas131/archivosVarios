var groupUsers = users.SelectMany(u => u.Value.Groups.Select(
    g => new
    {
        Group = g,
        User = u
    }
)).GroupBy(x => x.Group).ToDictionary(
    g => g.Key,
    g => g.Select(x => x.User).ToList()
);

using (var workbook = new XLWorkbook())
{
    var ws = workbook.Worksheets.Add("DevOps Project Users");
    ws.Cell(1, 1).Value = projectName;

    var range = ws.Range("A1:C1");
    range.Merge();
    range.Style.Alignment.Horizontal = XLAlignmentHorizontalValues.Center;

    ws.Cell(2, 1).Value = "Grupos";
    ws.Cell(2, 2).Value = "Nombres";
    //ws.Cell(2, 3).Value = "Email";

    int row = 3;

    foreach (var group in groupUsers)
    {
        ws.Cell(row, 1).Value = group.Key;

        var members = string.Join(", ", group.Value.Select(U => $"{U.Value.Name} - {U.Value.Email}"));
    
        ws.Cell(row, 2).Value = members;

        row++;
        
    }

    _loggerService.LogInfo($"Cantidad de Registros: {groupUsers.Count()}");
    _loggerService.LogInfo($"Cantidad de Colaboradores: {collaboratorCount}");

    string desktopPath = Environment.GetFolderPath(Environment.SpecialFolder.Desktop);

    string excelFileName = "DevOpsUsers - " + projectName + " - " + DateTime.Now.ToString("yyyy-MM-dd") + ".xlsx";

    ws.Columns().AdjustToContents();
    //workbook.SaveAs("C:\\Users\\afleitas004\\Desktop\\DevOpsUsers.xlsx");
    workbook.SaveAs(Path.Combine(desktopPath, excelFileName));

    _loggerService.LogInfo($"Archivo excel generado en el escritorio: {excelFileName} - {DateTime.Now}");

}
