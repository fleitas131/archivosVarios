def ejecutar_carga_contactos_salesforce():
    archivo_excel = entry_excel.get()
    correo_login = entry_correo.get()

    # Crear copia del Excel original
    archivo_resultado = archivo_excel.replace(".xlsx", "_resultado.xlsx")
    shutil.copy(archivo_excel, archivo_resultado)

    # Abrimos con openpyxl para conservar estilos
    workbook = load_workbook(archivo_resultado)
    sheet = workbook.active
    
    try:
        if not archivo_excel or not correo_login:
            raise Exception("Debes seleccionar un archivo Excel e ingresar un correo corporativo.")

        # Leemos con pandas (sin perder compatibilidad)
        df_resultado = pd.read_excel(archivo_resultado, engine='openpyxl')
        
        if 'PRID' not in df_resultado.columns:
            raise Exception("El archivo debe tener una columna llamada 'PRID'.")
        
        # ✅ Forzar tipos correctos para evitar el error float64
        df_resultado['PRID'] = df_resultado['PRID'].astype(str).str.replace('.0', '', regex=False)
        df_resultado['Estado'] = df_resultado['Estado'].astype(str)
        df_resultado['Empresa'] = df_resultado['Empresa'].astype(str)

        # ✅ Limpiar valores "nan" que Pandas pone en celdas vacías
        df_resultado['Estado'] = df_resultado['Estado'].replace('nan', '')
        df_resultado['Empresa'] = df_resultado['Empresa'].replace('nan', '')

        inicio = int(entry_inicio.get())
        fin = int(entry_fin.get())
        prids = df_resultado['PRID'].iloc[inicio - 1:fin].tolist()

        driver = GenerarBrowser()
        if not driver:
            raise Exception("No se pudo iniciar el navegador.")

        login_ok = login_pwc(driver, correo_login)
        if not login_ok:
            raise Exception("No se pudo realizar el login.")
        
        for idx, row in df_resultado.iloc[inicio - 1:fin].iterrows():
            prid = row["PRID"]
            log.insert(tk.END, f"Procesando fila {idx+1} - PRID: {prid}\n")
            log.see(tk.END)
            success = buscar_por_prid(driver, prid, row, df_resultado, idx, archivo_resultado, workbook, sheet)

            estado_actual = sheet.cell(row=idx + 2, column=9).value
            if estado_actual is None:
                sheet.cell(row=idx + 2, column=9).value = "Pendiente. Verificar log."
                estado_actual = "Pendiente. Verificar log."
            
            empresa_actual = sheet.cell(row=idx + 2, column=10).value  # Columna J (Empresa)
            if empresa_actual is None:
                sheet.cell(row=idx + 2, column=10).value = "Pendiente. Verificar log."
                empresa_actual = "Pendiente. Verificar log."

            log.insert(tk.END, f"Resultado PRID {prid}: {estado_actual} | Empresa: {empresa_actual}\n")
            log.see(tk.END)
            if not success:
                log.insert(tk.END, f"Error en PRID: {prid}\n")

        driver.quit()
        messagebox.showinfo("Completado", "Carga de Contactos por PRIDs finalizado.")
        logger.info("Carga de Contactos por PRIDs finalizado.")

    except Exception as e:
        messagebox.showerror("Error", str(e))
