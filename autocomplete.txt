import { Component, OnInit, Input, Output, EventEmitter, ViewChild, ElementRef} from '@angular/core';
import { FormGroup, FormControl, ControlValueAccessor, Validators } from '@angular/forms';
import { tap, Observable, of } from 'rxjs';
import { startWith } from 'rxjs/internal/operators/startWith';
import { debounceTime, distinctUntilChanged, map, switchMap } from 'rxjs/operators';
import { ChangeDetectorRef } from '@angular/core';
import { MatAutocompleteTrigger } from '@angular/material/autocomplete';

export interface Country {
  id: number;
  description: string;
}

@Component({
    selector: 'app-autocomplete-input',
    templateUrl: './autocomplete-input.component.html',
    styleUrl: './autocomplete-input.component.scss',
    standalone: false
})
export class AutocompleteInputComponent  implements OnInit {

  @Input() dataSource?: () => Observable<any[]>;
  @Input() paginatedSource?: (
    description: string,
    page:number,
    pageSize: number,
    countryId?: number,
  ) => Observable<any[]>;
  @Input() pageSize = 10;

  @Input() displayField: string;
  @Input() formGroup: FormGroup;
  @Input() label: string = '';
  @Input() controlName: string;

  @Input() loadOnFocus: boolean = false;

  @Output() itemChange: EventEmitter<Country> = new EventEmitter();

  @ViewChild(MatAutocompleteTrigger)
  autocompleteTrigger: MatAutocompleteTrigger;

  @ViewChild('inputElement') inputElement: ElementRef;
  
  options: any[] = [];
  filteredOptions: Observable<any[]>;
  private onChange = (value: any) => {};
  private onTouched = () => {};

  page = 0;
  hasMore = true;
  loading = false;
  currentSearch = '';
  typedValue: string = '';

  constructor(private cd: ChangeDetectorRef) {}

  get control(): FormControl {
    return this.formGroup.get(this.controlName) as FormControl;
  }

  ngOnInit(): void {
    const parentControl = this.formGroup.get(this.controlName);
    if (parentControl) {
      this.control.setValue(parentControl.value);
      parentControl.valueChanges.subscribe(value => {
        this.control.setValue(value, { emitEvent: false });
      });
    }

    //si no es paginado
    if (this.dataSource && !this.loadOnFocus) {
      this.dataSource().subscribe(data => {
        this.options = data;
        this.filteredOptions = this.control.valueChanges.pipe(
          startWith(''),
          map(value => {
            const filterValue =
            typeof value === 'string' 
            ? value 
            : this.resolveDisplay(value)/*value?.[this.displayField]*/
            return this._filter(filterValue || '');
          })
        );
      });
    }

    //si es paginado
    if (this.paginatedSource) {
      this.filteredOptions = this.control.valueChanges.pipe(
        debounceTime(300),
        map(value => typeof value === 'string'
            ? value
            : value?.[this.displayField] ?? ''
        ),
        distinctUntilChanged(),
        switchMap(value => {
          if (!value || value.trim().length === 0) {
            this.resetPagination();
            return of([]);
          }

          this.currentSearch = value;
          this.resetPagination();
          return this.loadPage();
        }
      ));
    }

    this.control.valueChanges.subscribe(value => {
      if (!value) {
        this.resetPagination();
      }
    })

    setTimeout(() => {
      const currentValue = this.control.value;
      if (currentValue) {
        this.control.setValue(currentValue);
      }
    })
  }

  onFocus() {
    if (this.dataSource && this.loadOnFocus) {
      this.dataSource().subscribe(data => {
        this.options = data;
        console.log(data);
        //this.control.setValue(this.control.value || '');
        
      });
    } 
    
    this.autocompleteTrigger.openPanel();
  }

  onOpened() {
    setTimeout(() => {
      const panel = document.querySelector('.mat-mdc-autocomplete-panel');
      if (!panel) return;
      panel.addEventListener('scroll', this.onScroll);
    });
  }

  onScroll = (event: any) => {
    const panel = event.target;
    const atBottom = panel.scrollHeight - panel.scrollTop <= panel.clientHeight + 10;
    if (atBottom && this.hasMore && !this.loading) {
      this.loadMore();
    }
  };

  private resolveDisplay(item: any): string {
    if (!item) return '';
    if (typeof item === 'string') return item;
    if (typeof item === 'number') {
      const found = this.options.find(opt => opt.id === item);
      return found ? found[this.displayField] : '';
    }
    return item[this.displayField];
  }

  displayFn = (item: any): string => {
    //return item ? item[this.displayField] : '';

    return this.resolveDisplay(item);
  };

  private _filter(value: string): any[] {
    if (!value) {
      return this.options;
    }

    const filterValue = value.toLowerCase();
    return this.options.filter(option => option[this.displayField].toLowerCase().includes(filterValue))
  }

  private loadPage(): Observable<any[]> {
    if (!this.paginatedSource) {
      return of(this.options);
    }
    
    this.loading = true;

    return this.paginatedSource(
      this.currentSearch,
      this.page,
      this.pageSize
    ).pipe(
      tap(result => {
        if (result.length < this.pageSize) {
          this.hasMore = false;
        }

        this.options = [...this.options, ...result];
        this.page++;
        this.loading = false;

        console.log(this.options);

        //this.filteredOptions = of(this.options);

        this.cd.detectChanges();
        this.autocompleteTrigger.updatePosition();

      }),
      map(() => this.options)
    );
  }

  loadMore() {
    if (this.loading || !this.hasMore) return;
    this.loadPage().subscribe();
  }

  private resetPagination() {
    this.page = 0;
    this.options = [];
    this.hasMore = true;
  }

  selectedOptionChanged(item: any) {
    if (!item) return;

    //const fullItem = this.options.find(opt => opt.id === item.id) || item;

    this.control.setValue(item);
    //this.autocompleteTrigger.writeValue(item[this.displayField]);
    this.itemChange.emit(item);
  }

  get isRequired(): boolean {
    return this.control.hasValidator(Validators.required) ?? false;
  }

      .subscribe((countryId) => {
        this.registeredStateSuggestService.countryId = countryId;
        this.ticketForm.get(['registeredAddress', 'stateId']).reset();
        setTimeout(() => {
          this.ticketForm.get(['registeredAddress', 'stateId']).markAsTouched();  
        }, 500);
      });
    
    // On operating address country change
    this.ticketForm
      .get(['operatingAddress', 'countryId'])
      .valueChanges
      .subscribe((countryId) => {
        this.operatingStateSuggestService.countryId = countryId;
        this.ticketForm.get(['operatingAddress', 'stateId']).reset();
        setTimeout(() => {
          this.ticketForm.get(['operatingAddress', 'stateId']).markAsTouched();
        }, 500);
      });
    
    //On PRID change
    const currentUrl = this.router.url;

    if (currentUrl.includes('/ems/entity/update/verification'))
    {
        this.ticketForm.get(['generalInformation', 'prid']).valueChanges.pipe(
        filter(() => !this.createMode && (this.ticketId <= 0)),
        startWith(''),
        distinctUntilChanged(),
        // delay emits
        debounceTime(1000),
        switchMap(prid => {
          if (prid !== '' && typeof prid == 'string') {
            this.loadingService.setSpinner(true);
            return this.entityService.getByPrid(prid, this.ticketTypeId);
          }
          else return of(null);
        }),
        filter(entity => entity != null)
      ).subscribe({
        next: entity => {
          if(entity?.generalInformation?.prid != '' && entity?.generalInformation?.prid != null){
            this.patchEntityByPrid(entity);
          }
        }
      });
    }
    

    // On Ticket status change
    this.status$.pipe(
      filter(s => s == Status.OPEN && (this.createMode || this.updateMode))
    ).subscribe(() => {
      this.copyAddressEnabled = false;
      this.ticketForm.disable({ emitEvent: false });
    })

    // No validation is performed in approve mode
    this.status$.pipe(
      filter(s => s == Status.OPEN && this.approveMode)
    ).subscribe(() => {
      this.clearEntityFormValidators()
    })

    // On any Registered form values changes
    // Enabled address copy if there's a value
    this.ticketForm.get('registeredAddress')
      .valueChanges
      .subscribe(() => {
        this.checkCopyAddressEnable();
      })

    console.log('stage', this.ticketStageDesc);
    console.log('ticket type', this.ticketTypeDesc);

    
    
    // On Party Id change
    this.ticketForm.get(['generalInformation', 'partyTypeId'])
      .valueChanges
      .subscribe(async (partyTypeId:number) => {
        const emailControl = this.ticketForm.get(['generalInformation', 'email']);
        const EXTERNAL_PERSON = 'External Person';

        const item = this.selectItem
          .generalInformation
          .partyTypeId
          .find(x => x.id == partyTypeId)

        if (item.description == EXTERNAL_PERSON && this.ticketStageDesc == 'Entity Update' && this.ticketTypeDesc == 'Entity Update') {
          emailControl.removeValidators(Validators.required);
        }
        else {
          emailControl.addValidators(Validators.required);
        }
        
        emailControl.updateValueAndValidity();

        // Updates service partyType for filtering organization types
        this.ticketForm.get(['generalInformation', 'organisationTypeId']).reset();
        this.ticketForm.get(['generalInformation', 'organisationTypeId']).markAsTouched();
        this.organizationTypeSuggestService.partyType = item.description;
        this.selectItem.generalInformation.organisationTypeId = await lastValueFrom(this.organizationTypeSuggestService.getAllItems());
      })


    if ((this.updateMode && (this.status == Status.PENDING) && this.ticketForm.get(['header','ticketId']).value > 0)) {
      this.ticketForm.get(['generalInformation', 'prid']).disable();
    }

    setTimeout(() => {
      this.markFormAsTouched();
    }, 1000);

    console.log(this.ticketForm);
  }

  navigateToHistoricalData(event: MatTabChangeEvent) {
    const stage = Stage[
      event.tab.textLabel
        .replace(/ /gm, '')
        .replace('/', '') as keyof typeof Stage
    ];
    
    let linkObj = null;
    
    switch (stage) {
      case Stage.HierarchyCreationUpdate: linkObj = [AppRoute.HIERARCHY, this.ticketId]; break;          
      case Stage.EntityCreation: 
        // if (this.status == Status.OPEN) {
        //   linkObj = [AppRoute.ENTITY, AppRoute.CREATE, AppRoute.APPROVE, this.ticketId];
        // } else {
          linkObj = [AppRoute.ENTITY, AppRoute.CREATE, this.ticketId]; 
        //}
        break;
      case Stage.EntityVerification: linkObj = [AppRoute.VERIFICATION, this.ticketId]; break;
      case Stage.EntityUpdate: 
        if (this.status == Status.OPEN) {
          linkObj = [AppRoute.ENTITY, AppRoute.UPDATE, AppRoute.APPROVE, this.ticketId];
        } else {
          linkObj = [AppRoute.ENTITY, AppRoute.UPDATE, this.ticketId]; 
        }
        break;
        
      default: linkObj = null; break;
    }
    
    linkObj.unshift(AppRoute.EMS);
    linkObj.unshift(AppRoute.ROOT);
    
    this.router.navigate(linkObj, {
      fragment: Stage[stage]
    });
  }

  applyContextValidators() {
    if (this.status == Status.OPEN) {
      this.clearEntityFormValidators();
      return;      
    }

    for (const subform in this.ticketForm.controls) {
      for (const controlKey in this.ticketForm.get(subform).value) {
        const validators = this.getContextValidators(subform, controlKey);

        if (validators != null) {
          const control = this.ticketForm.get([`${subform}`, `${controlKey}`]);
          control.clearValidators();          
          control.addValidators(validators);
          control.updateValueAndValidity({ emitEvent: false });
        }
      }
    }
  }

  getContextValidators(subformKey: string, controlKey: string): ValidatorFn[] { 
    const clientType = this.ticketForm.get(['header', 'clientType']).value as boolean;
    const ticketType = <TicketType>this.ticketTypeId;

    const validatorSwitch = (ignoreValue: boolean = false, requiredValue: boolean = true) => {
      return !ignoreValue && clientType == requiredValue ?
        [Validators.required] :
        [];
    };
    
    const requireValue = () => {
      return (ticketType == TicketType.CreateEntity) || (ticketType == TicketType.EntityVerification) || (ticketType == TicketType.SalesforceOpportunity) ?
        [Validators.required] :
        [];
    };

    const valueIgnored = [TicketType.UpdateEntity].includes(ticketType);

    const validators: any = {
      generalInformation: {
        prid: ((ticketType == TicketType.EntityVerification) || (ticketType == TicketType.UpdateEntity)) ? [Validators.required] : [],
        organisationTypeId: requireValue(),
        identifierTypeId: requireValue(),
        identifierValue: requireValue(),
        pwcSubindustryId: validatorSwitch(valueIgnored),
      },
      registeredAddress: {
        countryId: requireValue(),
        stateId: requireValue(),
        city: requireValue(),
        addressLine1: validatorSwitch(valueIgnored),
      },
      operatingAddress: {
        countryId: validatorSwitch(valueIgnored),
        stateId: validatorSwitch(valueIgnored),
        city: validatorSwitch(valueIgnored),
        addressLine1: validatorSwitch(valueIgnored),
      }
    }

    if (!(subformKey in validators)) return null;
    if (!((controlKey in validators[`${subformKey}`]))) return null;

    return validators[`${subformKey}`][`${controlKey}`];
  }

    const subIndustryId = subIndustry.id;
    this.ticketForm.get(['generalInformation','pwcSubindustryId']).setValue(subIndustryId);
  }

  async submit() {
    let ticket = this.getTicketFromForm();

    this.ticketForm.updateValueAndValidity();
    if (this.ticketForm.invalid) {
      this.toastrService.error(this.translate.instant('TOASTR.ENTITY-ERROR'))
      return;
    }
    
    const confirm = await lastValueFrom(
      this.dialogService.openConfirmDialog(this.translate.instant('TOASTR.CONFIRMATION-MESSAGE'), '').afterClosed()
    );
    if (!confirm) return;

    this.loadingService.setSpinner(true);
    const result = await lastValueFrom(
      this.ticketService.submit(ticket)
    );

    this.router.navigate(['/', AppRoute.OVERVIEW]);    
    this.ticketId = result.header.ticketId;
    this.status = result.header.status;

    //this.ticketForm.get('header').patchValue(result.header, { emitEvent: false });
    
  }

  async accept() {
    let ticket = this.getTicketFromForm();

    this.ticketForm.updateValueAndValidity();
    if (this.ticketForm.invalid) {
      this.toastrService.error(this.translate.instant('TOASTR.ENTITY-ERROR'))
      return;
    }

    const confirm = await lastValueFrom(
      this.dialogService.openConfirmDialog(this.translate.instant('TOASTR.APPROVAL-MESSAGE'), '').afterClosed()
    );
    if (!confirm) return;
   
    const sourceDesc = (ticket?.header?.sourceDesc == 'EMS Tool' && ticket.header.ticketTypeDesc != 'Entity Verification' ) ? 'EMS Tool' : 'Automatico';


    this.loadingService.setSpinner(true);
    var result = await lastValueFrom(
      this.ticketService.accept(ticket, sourceDesc)
    );

    if ((result.generalInformation.reason)) {
      this.toastrService.warning(result.generalInformation.reason);
    } else {
      this.router.navigate(['/', AppRoute.OVERVIEW]);
    }

    //this.router.navigate(['/', this.ticketId ? AppRoute.OVERVIEW : AppRoute.DASHBOARD]);
  }

  async reject() {
    let ticket = this.getTicketFromForm();

    this.ticketForm.updateValueAndValidity();
    // if (this.ticketForm.invalid) {
    //   this.toastrService.error(this.translate.instant('TOASTR.ENTITY-ERROR'))
    //   return;
    // }
    
    const dialogData = await lastValueFrom(
      this.dialogService.openRejectDialog(this.translate.instant('TOASTR.REJECT-MESSAGE'), '').afterClosed()
    );
    if (!dialogData.rejected) return;
    
    this.loadingService.setSpinner(true);
    const result = await lastValueFrom(
      this.ticketService.reject(ticket, dialogData.rejectionMessage)
    );

    this.router.navigate(['/', AppRoute.OVERVIEW]);
  }

  canApprove(): boolean {
    const permissions = [Permission.UpdateEntityApproval, Permission.CreateEntityApproval];
    return this.authService.isAuthorized(permissions);
  }  

  canSubmit(): boolean {
    const permission = this.createMode ? Permission.CreateEntityEdit :
      this.updateMode ? Permission.UpdateEntityEdit :
        null;
    const userTenantId = this.authService.currentUser.tenantId;
    

    return this.authService.isAuthorized(permission)
      //&& ((userTenantId == this.tenantId) || this.ticketId <= 0 || this.ticketId == null);
  }
  
  canSubmitEMSTeamMember(): boolean {
    const isRequester = this.authService.currentUser.roles.some(
      (role) => role.description == Role.Requestor,
    );
    const isImplementerHierachy = this.authService.currentUser.roles.some(
      (role) => role.description == Role.Implementer_Hierarchy,
    );
    const isIndependenceSpecialist = this.authService.currentUser.roles.some(
      (role) => role.description == Role.Independence_Specialist,
    );

    return (
      isRequester &&
      (isImplementerHierachy || isIndependenceSpecialist) &&
      (this.authService.currentUser.tenantId == 1 ||
        this.authService.currentUser.tenantId == 4)
    ); //Solo aplica para ARG o VEN
  }

    return (this.canApprove() && this.status == Status.OPEN && this.ticketStageDesc == 'Entity Creation')
