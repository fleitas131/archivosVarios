import { Component, OnInit, Input, Output, EventEmitter, ViewChild, ElementRef} from '@angular/core';
import { FormGroup, FormControl, ControlValueAccessor, Validators } from '@angular/forms';
import { tap, Observable, of } from 'rxjs';
import { startWith } from 'rxjs/internal/operators/startWith';
import { debounceTime, distinctUntilChanged, map, switchMap } from 'rxjs/operators';
import { ChangeDetectorRef } from '@angular/core';
import { MatAutocompleteTrigger } from '@angular/material/autocomplete';

export interface Country {
  id: number;
  description: string;
}

@Component({
    selector: 'app-autocomplete-input',
    templateUrl: './autocomplete-input.component.html',
    styleUrl: './autocomplete-input.component.scss',
    standalone: false
})
export class AutocompleteInputComponent  implements OnInit {

  @Input() dataSource?: () => Observable<any[]>;
  @Input() paginatedSource?: (
    description: string,
    page:number,
    pageSize: number,
    countryId?: number,
  ) => Observable<any[]>;
  @Input() pageSize = 1000;

  @Input() displayField: string;
  @Input() formGroup: FormGroup;
  @Input() label: string = '';
  @Input() controlName: string;

  @Input() loadOnFocus: boolean = false;

  @Output() itemChange: EventEmitter<Country> = new EventEmitter();

  @ViewChild(MatAutocompleteTrigger)
  autocompleteTrigger: MatAutocompleteTrigger;

  @ViewChild('inputElement') inputElement: ElementRef;
  
  options: any[] = [];
  filteredOptions: Observable<any[]>;
  private onChange = (value: any) => {};
  private onTouched = () => {};

  page = 0;
  hasMore = true;
  loading = false;
  currentSearch = '';
  typedValue: string = '';

  constructor(private cd: ChangeDetectorRef) {}

  get control(): FormControl {
    return this.formGroup.get(this.controlName) as FormControl;
  }

  ngOnInit(): void {
    const parentControl = this.formGroup.get(this.controlName);
    if (parentControl) {
      this.control.setValue(parentControl.value);
      parentControl.valueChanges.subscribe(value => {
        this.control.setValue(value, { emitEvent: false });
      });
    }

    //si no es paginado
    if (this.dataSource /*&& !this.loadOnFocus*/) {
      this.dataSource().subscribe(data => {
        this.options = data;
        this.filteredOptions = this.control.valueChanges.pipe(
          startWith(''),
          map(value => {
            const filterValue =
            typeof value === 'string' 
            ? value 
            : this.resolveDisplay(value)/*value?.[this.displayField]*/
            return this._filter(filterValue || '');
          })
        );
      });
    }

    //si es paginado
    if (this.paginatedSource) {
      this.filteredOptions = this.control.valueChanges.pipe(
        debounceTime(300),
        map(value => typeof value === 'string'
            ? value
            : value?.[this.displayField] ?? ''
        ),
        distinctUntilChanged(),
        switchMap(value => {
          if (!value || value.trim().length === 0) {
            this.resetPagination();
            return of([]);
          }

          this.currentSearch = value;
          this.resetPagination();
          return this.loadPage();
        }
      ));
    }

    this.control.valueChanges.subscribe(value => {
      // Si el input queda vacío, resetea paginación y vuelve a cargar si hay dataSource
      if (!value || (typeof value === 'string' && value.trim() === '')) {
        this.resetPagination();
        if (this.dataSource && !this.paginatedSource) {
          this.dataSource().subscribe(data => {
            this.options = data;
            this.cd.detectChanges();
          });
        }
      }
    })

    setTimeout(() => {
      const currentValue = this.control.value;
      if (currentValue) {
        this.control.setValue(currentValue);
      }
    })
  }

  onFocus() {
    if (this.dataSource /*&& this.loadOnFocus*/) {
      this.dataSource().subscribe(data => {
        this.options = data;
        console.log(data);

        this.filteredOptions = this.control.valueChanges.pipe(
          startWith(''),
          map(value => {
            const filterValue =
            typeof value === 'string' 
            ? value 
            : this.resolveDisplay(value)
            return this._filter(filterValue || '');
          })
        );

        this.cd.detectChanges();
        //this.control.setValue(this.control.value || '');
        
      });
    } 
    
    //this.autocompleteTrigger.openPanel();
  }

  onOpened() {
    setTimeout(() => {
      const panel = document.querySelector('.mat-mdc-autocomplete-panel');
      if (!panel) return;
      panel.addEventListener('scroll', this.onScroll);
    });
  }

  onScroll = (event: any) => {
    const panel = event.target;
    const atBottom = panel.scrollHeight - panel.scrollTop <= panel.clientHeight + 10;
    if (atBottom && this.hasMore && !this.loading) {
      this.loadMore();
    }
  };

  private resolveDisplay(item: any): string {
    if (!item) return '';
    if (typeof item === 'string') return item;
    if (typeof item === 'number') {
      const found = this.options.find(opt => opt.id === item);
      return found ? found[this.displayField] : '';
    }
    return item[this.displayField];
  }

  displayFn = (item: any): string => {
    //return item ? item[this.displayField] : '';

    return this.resolveDisplay(item);
  };

  private _filter(value: string): any[] {
    if (!value) {
      return this.options;
    }

    const filterValue = value.toLowerCase();
    return this.options.filter(option => option[this.displayField].toLowerCase().includes(filterValue))
  }

  private loadPage(): Observable<any[]> {
    if (!this.paginatedSource) {
      return of(this.options);
    }
    
    this.loading = true;

    return this.paginatedSource(
      this.currentSearch,
      this.page,
      this.pageSize
    ).pipe(
      tap(result => {
        if (result.length < this.pageSize) {
          this.hasMore = false;
        }

        this.options = [...this.options, ...result];
        this.page++;
        this.loading = false;

        console.log(this.options);

        //this.filteredOptions = of(this.options);

        this.cd.detectChanges();
        this.autocompleteTrigger.updatePosition();

      }),
      map(() => this.options)
    );
  }

  loadMore() {
    if (this.loading || !this.hasMore) return;
    this.loadPage().subscribe();
  }

  private resetPagination() {
    this.page = 0;
    this.options = [];
    this.hasMore = true;
  }

  selectedOptionChanged(item: any) {
    if (!item) return;

    this.control.setValue(item, { emitEvent: false });
    this.itemChange.emit(item);
    this.autocompleteTrigger.closePanel();
  }

  get isRequired(): boolean {
    return this.control.hasValidator(Validators.required) ?? false;
  }

}

      this.cdr.detectChanges(); // <-- fuerza la actualización

    const ticketHistory = this.activatedRoute.snapshot.data['ticketHistory'] as Ticket[];

    //console.log('ticket history', ticketHistory);

    if (!Array.isArray(ticketHistory) || ticketHistory.length === 0) {
      return;

      const historicalTicketForm = this.generateTicketForm(true);

      historicalTicketForm.get('header').patchValue(item.header, { emitEvent: false }); 
      historicalTicketForm.get('generalInformation').patchValue(item.generalInformation, { emitEvent: false }); 
      historicalTicketForm.get('operatingAddress').patchValue(item.operatingAddress, { emitEvent: false }); 
      historicalTicketForm.get('registeredAddress').patchValue(item.registeredAddress, { emitEvent: false }); 

      historicalTicketForm.disable({ emitEvent: false });

      this.ticketFormHistory.push(historicalTicketForm);
    }

  navigateToHistoricalData(event: MatTabChangeEvent) {
    const stage = Stage[
      event.tab.textLabel
        .replace(/ /gm, '')
        .replace('/', '') as keyof typeof Stage
    ];
    
    let linkObj = null;
    
    switch (stage) {
      case Stage.HierarchyCreationUpdate: linkObj = [AppRoute.HIERARCHY, this.ticketId]; break;          
      case Stage.EntityCreation: 
        // if (this.status == Status.OPEN) {
        //   linkObj = [AppRoute.ENTITY, AppRoute.CREATE, AppRoute.APPROVE, this.ticketId];
        // } else {
          linkObj = [AppRoute.ENTITY, AppRoute.CREATE, this.ticketId]; 
        //}
        break;
      case Stage.EntityVerification: linkObj = [AppRoute.VERIFICATION, this.ticketId]; break;
      case Stage.EntityUpdate: 
        if (this.status == Status.OPEN) {
          linkObj = [AppRoute.ENTITY, AppRoute.UPDATE, AppRoute.APPROVE, this.ticketId];
        } else {
          linkObj = [AppRoute.ENTITY, AppRoute.UPDATE, this.ticketId]; 
        }
        break;
        
      default: linkObj = null; break;
    }
    
    linkObj.unshift(AppRoute.EMS);
    linkObj.unshift(AppRoute.ROOT);
    
    this.router.navigate(linkObj, {
      fragment: Stage[stage]
    });
  }

  applyContextValidators() {
    if (this.status == Status.OPEN) {
      this.clearEntityFormValidators();
      return;      
    }

    for (const subform in this.ticketForm.controls) {
      for (const controlKey in this.ticketForm.get(subform).value) {
        const validators = this.getContextValidators(subform, controlKey);

        if (validators != null) {
          const control = this.ticketForm.get([`${subform}`, `${controlKey}`]);
          control.clearValidators();          
          control.addValidators(validators);
          control.updateValueAndValidity({ emitEvent: false });
        }
      }
    }
  }

  getContextValidators(subformKey: string, controlKey: string): ValidatorFn[] { 
    const clientType = this.ticketForm.get(['header', 'clientType']).value as boolean;
    const ticketType = <TicketType>this.ticketTypeId;

    const validatorSwitch = (ignoreValue: boolean = false, requiredValue: boolean = true) => {
      return !ignoreValue && clientType == requiredValue ?
        [Validators.required] :
        [];
    };
    
    const requireValue = () => {
      return (ticketType == TicketType.CreateEntity) || (ticketType == TicketType.EntityVerification) || (ticketType == TicketType.SalesforceOpportunity) || (ticketType == TicketType.EntityCreationVerifcation) ?
        [Validators.required] :
        [];
    };

    const valueIgnored = [TicketType.UpdateEntity].includes(ticketType);

    const validators: any = {
      generalInformation: {
        prid: ((ticketType == TicketType.EntityVerification) || (ticketType == TicketType.UpdateEntity)) ? [Validators.required] : [],
        organisationTypeId: requireValue(),
        identifierTypeId: requireValue(),
        identifierValue: requireValue(),
        pwcSubindustryId: validatorSwitch(valueIgnored),
      },
      registeredAddress: {
        countryId: requireValue(),
        stateId: requireValue(),
        city: requireValue(),
        addressLine1: validatorSwitch(valueIgnored),
      },
      operatingAddress: {
        countryId: validatorSwitch(valueIgnored),
        stateId: validatorSwitch(valueIgnored),
        city: validatorSwitch(valueIgnored),
        addressLine1: validatorSwitch(valueIgnored),
      }
    }

    if (!(subformKey in validators)) return null;
    if (!((controlKey in validators[`${subformKey}`]))) return null;

    return validators[`${subformKey}`][`${controlKey}`];
  }

  async submit() {

    this.ticketForm.updateValueAndValidity({ emitEvent: false });
    if (this.ticketForm.invalid) {
      this.toastrService.error(this.translate.instant('TOASTR.ENTITY-ERROR'))
      return;
    }

      this.dialogService.openConfirmDialog(this.translate.instant('TOASTR.CONFIRMATION-MESSAGE'), '').afterClosed()
    );
    if (!confirm) return;

    this.updateAutocompleteValues();

    this.loadingService.setSpinner(true);
    const result = await lastValueFrom(
      this.ticketService.submit(ticket)
    );

    this.router.navigate(['/', AppRoute.OVERVIEW]);    
    this.ticketId = result.header.ticketId;
    this.status = result.header.status;

    //this.ticketForm.get('header').patchValue(result.header, { emitEvent: false });
    
  }

  async accept() {
    let ticket = this.getTicketFromForm();

    this.ticketForm.updateValueAndValidity();
    if (this.ticketForm.invalid) {
      this.toastrService.error(this.translate.instant('TOASTR.ENTITY-ERROR'))
      return;
    }

    const confirm = await lastValueFrom(
      this.dialogService.openConfirmDialog(this.translate.instant('TOASTR.APPROVAL-MESSAGE'), '').afterClosed()
    );
    if (!confirm) return;
   
    const sourceDesc = (ticket?.header?.sourceDesc == 'EMS Tool' && (ticket.header.ticketTypeDesc != 'Entity Verification') && (ticket.header.ticketTypeDesc != 'Entity Creation & Verification') )
     ? 'EMS Tool' : 'Automatico';


    this.loadingService.setSpinner(true);
    var result = await lastValueFrom(
      this.ticketService.accept(ticket, sourceDesc)
    );

    if ((result.generalInformation.reason)) {
      this.toastrService.warning(result.generalInformation.reason);
    } else {
      this.router.navigate(['/', AppRoute.OVERVIEW]);
    }

    //this.router.navigate(['/', this.ticketId ? AppRoute.OVERVIEW : AppRoute.DASHBOARD]);
  }

  private restoreAutocompleteFields(entiy: IEntity): void {
    //registered country
    //const registeredCountryId = this.ticketForm.get(['registeredAddress', 'countryId']).value;
   if (
        entiy.registeredAddress &&
        Object.values(entiy.registeredAddress).some(v => v !== null && v !== undefined && v !== '')
      ) {
      const registeredCountryId = entiy.registeredAddress.countryId;
      if (registeredCountryId) {
        this.registeredCountrySuggestService.getSuggestPaginated(registeredCountryId.toString(), 0, 1)
        .subscribe(items => {
          const found = items.find(x =>x.id === registeredCountryId);
          if (found) {
            this.ticketForm.get(['registeredAddress', 'countryId']).setValue(found, { emitEvent: false });
          }
        })
      }

      //registered state
      const registeredStateId = this.ticketForm.get(['registeredAddress', 'stateId']).value;
      if (registeredStateId) {
        this.registeredStateSuggestService.getSuggestPaginated(registeredStateId.toString(), 0, 1, registeredCountryId)
        .subscribe(items => {
          const found = items.find(x =>x.id === registeredStateId);
          if (found) {
            this.ticketForm.get(['registeredAddress', 'stateId']).setValue(found, { emitEvent: false });
          }
        })
      }
    }

  if (
        entiy.operatingAddress &&
        Object.values(entiy.operatingAddress).some(v => v !== null && v !== undefined && v !== '')
      ) {
      //operating country
      const operatingCountryId = this.ticketForm.get(['operatingAddress', 'countryId']).value;
      if (operatingCountryId) {
        this.operatingCountrySuggestService.getSuggestPaginated(operatingCountryId.toString(), 0, 1)
        .subscribe(items => {
          const found = items.find(x =>x.id === operatingCountryId);
          if (found) {
            this.ticketForm.get(['operatingAddress', 'countryId']).setValue(found, { emitEvent: false });
          }
        })
      }

      //operating state
      const operatingStateId = this.ticketForm.get(['operatingAddress', 'stateId']).value;
      if (operatingStateId) {
        this.operatingStateSuggestService.getSuggestPaginated(operatingStateId.toString(), 0, 1, operatingCountryId)
        .subscribe(items => {
          const found = items.find(x =>x.id === operatingStateId);
          if (found) {
            this.ticketForm.get(['operatingAddress', 'stateId']).setValue(found, { emitEvent: false });
          }
        })
      }
    }

    //identifier
    const identifierId = this.ticketForm.get(['generalInformation','identifierTypeId']).value;
    if (identifierId) {
      this.identifierTypeSuggestService.getSuggestPaginated(identifierId.toString(), 0, 1)
      .subscribe(items => {
        const found = items.find(x =>x.id === identifierId);
        if (found) {
          this.ticketForm.get(['generalInformation','identifierTypeId']).setValue(found, { emitEvent: false });
        }
      })
    }

    //subindustry
    const subindustryId = this.ticketForm.get(['generalInformation','pwcSubindustryId']).value;
    if (subindustryId) {
      this.subindustrySuggestService.getSuggestPaginated(subindustryId.toString(), 0, 1)
      .subscribe(items => {
        const found = items.find(x =>x.id === subindustryId);
        if (found) {
          this.ticketForm.get(['generalInformation','pwcSubindustryId']).setValue(found, { emitEvent: false });
        }
      })
    }
  }

    });
  }

  //prid solo acepta numeros
  numbersOnly(event: Event): void {
    const input = event.target as HTMLInputElement;
    let cleaned = input.value.replace(/\D+/g, '');
    cleaned = cleaned.slice(0,9);
    input.value = cleaned;
    this.hierarchyForm.get(['generalInformation', 'prid']).setValue(cleaned, { emitEvent: false });
  }

  //vemos si tiene user asignado
  hasEMSUser(): boolean{
    let check: boolean = false;
    
    const assigneeEmail = this.hierarchyForm.get(['header', 'assignee']).value;
    const emsUserEmail = this.authService.currentUser.email;

    if (assigneeEmail != null && assigneeEmail == emsUserEmail && this.canApprove()) {
      check = true;
    }

    return check;
  }

  //si es ems user puede cancelar el ticket
  isNotRequester(): boolean{
    let check: boolean = false;
    
    const isNotReq = this.authService.currentUser.roles[0].id;

    if (isNotReq != 1 && this.hasEMSUser())
    {
      this.approveMode = true;
      check = true;
    }

    return check;
  }

  //CES & Hierarchy toggle exclusion
  private setMutualExclusion()
  {
    const entityCtrl = this.hierarchyForm.get(['generalInformation', 'entityWithoutHierarchy']);
    const cesCtrl = this.hierarchyForm.get(['generalInformation', 'cesRelationshipUpdated']);

    entityCtrl?.valueChanges.subscribe(value => {
      if (value === true)
      {
        cesCtrl?.setValue(false, { emitEvent: false });
      }
    });
    
    cesCtrl?.valueChanges.subscribe(value => {
      if (value === true)
      {
        entityCtrl?.setValue(false, { emitEvent: false });
      }
    });
  }

  //submit toggle disable control
  toggleSubmitDisable()
  {
    const entityCtrl = this.hierarchyForm.get(['generalInformation', 'entityWithoutHierarchy']);
    const cesCtrl = this.hierarchyForm.get(['generalInformation', 'cesRelationshipUpdated']);
    let submitDisable: boolean = false

    if (!entityCtrl.value && !cesCtrl.value)
    {
      submitDisable = true;
    }

    return submitDisable;
  }

  get pridToolTip(): string | null {

    let tipMessage: string = "";

    if (this.status == 1)
    {
      tipMessage = "Buscar en CES el PRID.\nEjemplo:\nPRID: 230618\nParty Name: IBM ARGENTINA SRL";
    }
    
    return tipMessage;
  }

  isHistoricaModeAvailable(): boolean {
    return [
      TicketType.SalesforceOpportunity,
      TicketType.EntityVerification
    ].includes(this.ticketTypeId)
  }

  navigateToHistoricalData(event: MatTabChangeEvent) {
    const stage = Stage[
      event.tab.textLabel
        .replace(/ /gm, '')
        .replace('/', '') as keyof typeof Stage
    ];
  
    let linkObj = null;
  
    switch (stage) {
      case Stage.EntityCreation: linkObj = [AppRoute.ENTITY, AppRoute.CREATE, this.ticketId]; break;
      case Stage.EntityUpdate: linkObj = [AppRoute.ENTITY, AppRoute.UPDATE, this.ticketId]; break;
      case Stage.EntityVerification: linkObj = [AppRoute.VERIFICATION, this.ticketId]; break;
      
      default: linkObj = null; break;
    }
  
    linkObj.unshift(AppRoute.EMS);
    linkObj.unshift(AppRoute.ROOT);
  
    this.router.navigate(linkObj, {
      fragment: Stage[stage]
    });
  }  
 
  patchHierarchy(hierarchy: IHierarchy) {
    if (!hierarchy) return;

    if (hierarchy.meta?.outputEntityMessage) {
      this.toastrService.warning(hierarchy.meta.outputEntityMessage);
      return;
    }

    this.hierarchyForm.get(['generalInformation']).patchValue(hierarchy.generalInformation, { emitEvent: false });

    this.markFormAsTouched();
  }

  patchTicket(ticket: Ticket) {
    if (!ticket) return;
    this.patchHierarchy(ticket);

    this.ticketId = ticket.header.ticketId;
    this.status = ticket.header.status;
    this.ticketTypeDesc = ticket.header.ticketTypeDesc;
    this.ticketStageDesc = ticket.header.ticketStageDesc;
    this.ticketTypeId = ticket.header.ticketTypeId;

  

    this.tenantId = ticket.header.tenantId;
    this.hierarchyForm.get('header').patchValue(ticket.header, { emitEvent: false });

    if(ticket.header.ticketTypeDesc == 'Salesforce Opportunity' || ticket.header.ticketTypeDesc == 'Entity Verification' || ticket.header.ticketTypeDesc == 'Entity Creation & Verification'){
      this.showToggle = true;
      this.requiredAttachmentCount = ticket.generalInformation.entityWithoutHierarchy ? 0 : 1;
      

    }

    this.hierarchyForm.get(['generalInformation', 'cesRelationshipUpdated']).patchValue(ticket.generalInformation.cesRelationshipUpdated, { emitEvent: false });
  }

    private markFormAsTouched(formGroup: FormGroup = this.hierarchyForm) {
    (<any>Object).values(formGroup.controls).forEach((control: any) => {
      if (control.controls) {
        this.markFormAsTouched(control);
      }
      else {
        control.markAsTouched();
      }
    });
  }

  getSubform(subformName: string | string[]): FormGroup {
    return <FormGroup>this.hierarchyForm.get(subformName);
  }

  loadSnapshotData() {
    this.ticketSourceId = this.activatedRoute.snapshot.data['entitySourceId'] as number;
    this.ticketTypeId = this.activatedRoute.snapshot.data['ticketTypeId'] as number;

    const ticket = this.activatedRoute.snapshot.data['ticket'] as Ticket;

    if (!!ticket) this.patchTicket(ticket);
    else this.hierarchyForm.get(['header', 'requestor']).patchValue(this.authService.currentUser.email);


    // SELECT ITEMS
    this.selectMultipleItem.generalInformation.entityTypeId = this.activatedRoute.snapshot.data['entityTypes'] as Item[] | [];
  }

  buildHistoricalDataForm() {
    const ticketHistory = this.activatedRoute.snapshot.data['ticketHistory'] as Ticket[];
    
    for (const item of ticketHistory) {
      const historicalTicketForm = this.generateTicketForm(true);
    
      historicalTicketForm.get('header').patchValue(item.header, { emitEvent: false });
      historicalTicketForm.get('generalInformation').patchValue(item.generalInformation, { emitEvent: false });
          
      historicalTicketForm.disable({ emitEvent: false });
    
      this.ticketFormHistory.push(historicalTicketForm);
    }
  }


  setMode() {
    this.createUpdateMode = !!(this.activatedRoute.snapshot.data[
      'createUpdateMode'
    ] as boolean);
    this.approveMode = !!(this.activatedRoute.snapshot.data[
      'approveMode'
    ] as boolean);
  }

  showTicketHistory() {
    this.dialogService.openTicketHistoryDialog(this.ticketId);

  createNewEntity() {
    //this.router.navigate(['/', AppRoute.EMS, AppRoute.ENTITY, AppRoute.CREATE]);
    const url = this.router.serializeUrl(this.router.createUrlTree([`/${AppRoute.EMS}/${AppRoute.ENTITY}/${AppRoute.CREATE}`]))
    window.open(url,'_blank')
  }

  close() {
    this.loadingService.setSpinner(true);
    this.router.navigate(['/', this.ticketId ? AppRoute.OVERVIEW : AppRoute.DASHBOARD]);
  }

  async cancel() {
    const confirm = await lastValueFrom(
      this.dialogService.openConfirmDialog(this.translate.instant('TOASTR.CANCELATION-MESSAGE'), '').afterClosed()
    );
    if (!confirm) return;

    this.loadingService.setSpinner(true);
    const result = await lastValueFrom(
      this.hierarchyService.cancel(this.ticketId)
    );

    this.router.navigate(['/', AppRoute.OVERVIEW]);
  }

  //nuevo cancel con comentario
  async newCancel()
  {
    let ticket = this.getTicketFromForm();

    this.hierarchyForm.updateValueAndValidity();
    // if (this.ticketForm.invalid) {
    //   this.toastrService.error(this.translate.instant('TOASTR.ENTITY-ERROR'))
    //   return;
    // }
    
    const dialogData = await lastValueFrom(
      this.dialogService.openCancelDialog(this.translate.instant('TOASTR.CANCELATION-MESSAGE'), '').afterClosed() 
    );
    if (!dialogData.cancelled) return;
    
    this.loadingService.setSpinner(true);
    const result = await lastValueFrom(
      this.hierarchyService.newCancel(this.ticketId, dialogData.cancelMessage)
    );

    this.router.navigate(['/', AppRoute.OVERVIEW]);
  }

  private getTicketFromForm(): Ticket {
    let ticket = Object.assign(
      new Ticket(),
      this.hierarchyForm.getRawValue()
    ) as Ticket;

    ticket.header.sourceId = this.ticketSourceId;
    ticket.header.ticketTypeId = this.ticketTypeId;
    ticket.header.fileIds = this.attachmentFileIds;
    ticket.header.commentIds = this.commentIds;

    return ticket;
  }

  updateAttachmentFileIds(fileIds: number[]) {
    this.attachmentFileIds = fileIds;

    if (this.status == this.TicketStatus.PENDING) {
      // ATTACHMENT COUNT CHECK

    }
      

  }

  updateCommentIds(commentsIds: number[]) {
    this.commentIds = commentsIds;
  }

  async submit() {
    let ticket = this.getTicketFromForm();

    this.hierarchyForm.updateValueAndValidity();
    if (this.hierarchyForm.invalid) {
      this.toastrService.error(this.translate.instant('TOASTR.ENTITY-ERROR'))
      return;
    }
    
    const confirm = await lastValueFrom(
      this.dialogService.openConfirmDialog(this.translate.instant('TOASTR.CONFIRMATION-MESSAGE'), '').afterClosed()
    );
    if (!confirm) return;

    this.loadingService.setSpinner(true);
    const result = await lastValueFrom(
      this.hierarchyService.submit(ticket)
    );
   
    this.ticketId = result.header.ticketId;
    this.status = result.header.status;
    
    this.hierarchyForm.get('header').patchValue(result.header, { emitEvent: false });
    this.router.navigate(['/', AppRoute.OVERVIEW]);
  }

  async accept() {
    let ticket = this.getTicketFromForm();

    this.hierarchyForm.updateValueAndValidity();
    if (this.hierarchyForm.invalid) {
      this.toastrService.error(this.translate.instant('TOASTR.ENTITY-ERROR'))
      return;
    }
    
    const confirm = await lastValueFrom(
      this.dialogService.openConfirmDialog(this.translate.instant('TOASTR.APPROVAL-MESSAGE'), '').afterClosed()
    );
    if (!confirm) return;
   
    const sourceDesc = 
    (this.ticketTypeDesc == 'Salesforce Opportunity' || this.ticketTypeDesc == 'Entity Verification' || this.ticketTypeDesc == 'Entity Creation & Verification') 
    ? 'Automatico' : 'EMS Tool';

    this.loadingService.setSpinner(true);
    const result = await lastValueFrom(
      this.hierarchyService.accept(ticket, sourceDesc)
    );

    this.router.navigate(['/', AppRoute.OVERVIEW]);
  }

  async reject() {
    let ticket = this.getTicketFromForm();

    this.hierarchyForm.updateValueAndValidity();
    // if (this.hierarchyForm.invalid) {
    //   this.toastrService.error(this.translate.instant('TOASTR.ENTITY-ERROR'))
    //   return;
    // }
    
    const dialogData = await lastValueFrom(
      this.dialogService.openRejectDialog(this.translate.instant('TOASTR.REJECT-MESSAGE'), '').afterClosed()
    );
    if (!dialogData.rejected) return;
    
    this.loadingService.setSpinner(true);
    const result = await lastValueFrom(
      this.hierarchyService.reject(ticket, dialogData.rejectionMessage)
    );

    this.router.navigate(['/', AppRoute.OVERVIEW]);
  }

  canApprove(): boolean {
    const permissions = [Permission.HierarchyApproval];
    const stage = this.hierarchyForm.get(["header", "stage"]).value as Stage;

    return this.authService.isAuthorized(permissions) && stage == Stage.HierarchyCreationUpdate;
  }  

  canSubmit(): boolean {
    return this.authService.isAuthorized(Permission.HierarchyEdit) 
    //&& ((userTenantId == userTenantId) || this.ticketId <= 0 || this.ticketId == null);
  }  

  test() {

    console.log("toggleSubmitDisable", this.toggleSubmitDisable());
    console.log("disableSubmit", this.disableSubmit()); 
  }

  disableSubmit(): boolean {
    const disable = (this.hierarchyForm.invalid ||
            ((this.attachmentFileIds.length < this.requiredAttachmentCount) && (this.toggleSubmitDisable() === true && this.showToggle === true)) || 
            ((this.attachmentFileIds.length < this.requiredAttachmentCount) && this.ticketTypeId == 4) 
          )
          return disable;           
  }

  //         )


}
